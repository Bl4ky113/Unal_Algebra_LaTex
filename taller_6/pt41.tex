\item Use las propiedades de determinantes para probar que \\
    \[
        \begin{vmatrix}
            a + b & a & a & a \\ 
            a & a + b & a & a \\ 
            a & a & a + b & a \\ 
            a & a & a & a + b \\ 
        \end{vmatrix}
        =
        b^3(4a + b)
    \]
    \[
        \begin{vmatrix}
            1 + a & b & c & d & e \\
            a & 1 + b & c & d & e \\
            a & b & 1 + c & d & e \\
            a & b & c & 1 + d & e \\
            a & b & c & d & 1 + e \\
        \end{vmatrix}
        = 
        1 + a + b + c + d + e
    \]
